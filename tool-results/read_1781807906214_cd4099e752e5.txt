    54→// ─── Helper: Transform R2 keys to public URLs ───
    55→function toPublicUrl(env: Env, url: string | null | undefined, bucket: string = 'images'): string {
    56→  if (!url) return '';
    57→  // Already a full URL (http/https) — return as-is
    58→  if (url.startsWith('http://') || url.startsWith('https://')) return url;
    59→  // R2 key — convert to public URL
    60→  return getPublicUrl(env, bucket, url);
    61→}

    65→// GET /config — Get server config for student app (public)
    66→routes.get('/config', async (c) => {
    67→  try {
    68→    const cachedConfig = await c.env.KV_CONFIG.get('server_config', 'json');
    69→    if (cachedConfig) {
    70→      const config = cachedConfig as ServerConfig;
    71→      return c.json({ config: transformConfigForStudent(config) });
    72→    }
    73→
    74→    const { results } = await c.env.DB.prepare(
    75→      'SELECT key, value FROM app_config'
    76→    ).all<{ key: string; value: string }>();
    77→
    78→    const configMap: Record<string, unknown> = {};
    79→    for (const row of results) {
    80→      try {
    81→        configMap[row.key] = JSON.parse(row.value);
    82→      } catch {
    83→        configMap[row.key] = row.value;
    84→      }
    85→    }
    86→
    87→    const config: ServerConfig = {
    88→      featureToggles: { ...DEFAULT_CONFIG.featureToggles, ...(configMap.featureToggles as Partial<ServerConfig['featureToggles']>) },
    89→      homePageSections: (configMap.homePageSections as ServerConfig['homePageSections']) || DEFAULT_CONFIG.homePageSections,
    90→      sidebarVisibility: { ...DEFAULT_CONFIG.sidebarVisibility, ...(configMap.sidebarVisibility as Partial<ServerConfig['sidebarVisibility']>) },
    91→      bottomNavTabs: (configMap.bottomNavTabs as ServerConfig['bottomNavTabs']) || DEFAULT_CONFIG.bottomNavTabs,
    92→      topBarElements: { ...DEFAULT_CONFIG.topBarElements, ...(configMap.topBarElements as Partial<ServerConfig['topBarElements']>) },
    93→      cardStyle: (configMap.cardStyle as ServerConfig['cardStyle']) || DEFAULT_CONFIG.cardStyle,
    94→      contentProtection: { ...DEFAULT_CONFIG.contentProtection, ...(configMap.contentProtection as Partial<ServerConfig['contentProtection']>) },
    95→    };
    96→
    97→    await c.env.KV_CONFIG.put('server_config', JSON.stringify(config), { expirationTtl: 300 });
    98→
    99→    return c.json({ config: transformConfigForStudent(config) });
   100→  } catch (error) {
   101→    return c.json({ config: transformConfigForStudent(DEFAULT_CONFIG) });
   102→  }

   216→// ─── Course Packages ───
   217→
   218→routes.get('/course-packages', async (c) => {
   219→  try {
   220→    const courseId = c.req.query('courseId');
   221→    if (!courseId) {
   222→      return c.json({ error: 'courseId required' }, 400);
   223→    }
   224→
   225→    // Get course price
   226→    const course = await c.env.DB.prepare(
   227→      'SELECT id, price FROM courses WHERE id = ? AND is_published = 1'
   228→    ).bind(courseId).first<{ id: string; price: number }>();
   229→
   230→    if (!course) {
   231→      return c.json({ packages: [] });
   232→    }
   233→
   234→    let result = await c.env.DB.prepare(
   235→      'SELECT * FROM course_packages WHERE course_id = ? AND is_active = 1 ORDER BY price ASC'
   236→    ).bind(courseId).all();
   237→
   238→    // Auto-generate Single & Duo packages if missing (always show both for paid courses)
   239→    if (course.price > 0) {
   240→      const existingTypes = new Set((result.results as any[]).map((p: any) => p.package_type));
   241→      let needsRefetch = false;
   242→
   243→      // Ensure Single package exists
   244→      if (!existingTypes.has('single')) {
   245→        await c.env.DB.prepare(`
   246→          INSERT INTO course_packages (course_id, package_type, price, duration_months, max_users, is_auto_assign, is_active, display_name, description)
   247→          VALUES (?, 'single', ?, 6, 1, 1, 1, 'Single', '1 জন ইউজারের জন্য')
   248→        `).bind(courseId, course.price).run();
   249→        needsRefetch = true;
   250→      }
   251→
   252→      // Ensure Duo package exists (original price + 15% extra)
   253→      if (!existingTypes.has('dual')) {
   254→        const duoPrice = Math.round(course.price * 1.15);
   255→        await c.env.DB.prepare(`
   256→          INSERT INTO course_packages (course_id, package_type, price, duration_months, max_users, is_auto_assign, is_active, display_name, description)
   257→          VALUES (?, 'dual', ?, 6, 2, 1, 1, 'Duo', '2 জন ইউজারের জন্য — বন্ধুকে শেয়ার করুন!')
   258→        `).bind(courseId, duoPrice).run();
   259→        needsRefetch = true;
   260→      }
   261→
   262→      // Re-fetch after auto-generation
   263→      if (needsRefetch) {
   264→        result = await c.env.DB.prepare(
   265→          'SELECT * FROM course_packages WHERE course_id = ? AND is_active = 1 AND package_type IN ("single", "dual") ORDER BY price ASC'
   266→        ).bind(courseId).all();
   267→      }
   268→    }
   269→
   270→    // Filter to only show Single and Duo packages (hide any other types)
   271→    const filteredPackages = (result.results as any[]).filter((p: any) => p.package_type === 'single' || p.package_type === 'dual');
   272→
   273→    return c.json({ packages: filteredPackages });
   274→  } catch (error) {
   275→    return c.json({ error: getErrorMessage(error) }, 500);
   276→  }
   277→});

   340→routes.get('/courses', async (c) => {
   341→  try {
   342→    const technology = c.req.query('technology') || '';
   343→    const limit = parseInt(c.req.query('limit') || '20');
   344→    const offset = parseInt(c.req.query('offset') || '0');
   345→    const search = c.req.query('search') || '';
   346→    const level = c.req.query('level') || '';
   347→
   348→    let where = 'WHERE is_published = 1';
   349→    const params: unknown[] = [];
   350→
   351→    if (technology) {
   352→      where += ' AND technology_id = ?';
   353→      params.push(technology);
   354→    }
   355→    if (search) {
   356→      where += ' AND title LIKE ?';
   357→      params.push(`%${search}%`);
   358→    }
   359→    if (level) {
   360→      where += ' AND level = ?';
   361→      params.push(level);
   362→    }
   363→
   364→    const countResult = await c.env.DB.prepare(
   365→      `SELECT COUNT(*) as total FROM courses ${where}`
   366→    ).bind(...params).first();
   367→    const total = (countResult as any)?.total || 0;
   368→
   369→    const result = await c.env.DB.prepare(
   370→      `SELECT * FROM courses ${where} ORDER BY created_at DESC LIMIT ? OFFSET ?`
   371→    ).bind(...params, limit, offset).all();
   372→
   373→    // Enrich courses with auto-calculated duration from videos
   374→    const enrichedCourses = await Promise.all(result.results.map(async (course: any) => {
   375→      try {
   376→        const vidStats = await c.env.DB.prepare(
   377→          'SELECT COUNT(*) as count, COALESCE(SUM(duration), 0) as total_duration FROM videos WHERE course_id = ?'
   378→        ).bind(course.id).first();
   379→        const vc = (vidStats as any)?.count || 0;
   380→        const td = (vidStats as any)?.total_duration || 0;
   381→        const avg = vc > 0 ? Math.round(td / vc * 10) / 10 : 0;
   382→        return {
   383→          ...course,
   384→          duration: avg,
   385→          total_videos: vc,
   386→          total_video_duration: td,
   387→          thumbnail_url: toPublicUrl(c.env, course.thumbnail_url, 'images'),
   388→        };
   389→      } catch {
   390→        return {
   391→          ...course,
   392→          thumbnail_url: toPublicUrl(c.env, course.thumbnail_url, 'images'),
   393→        };
   394→      }
   395→    }));
   396→
   397→    return c.json({ courses: enrichedCourses, total });
   398→  } catch (error) {
   399→    return c.json({ error: getErrorMessage(error) }, 500);
   400→  }
   401→});

   575→  title: string;
   576→  description: string | null;
   577→  technology_id: string | null;
   578→  instructor_id: string | null;
   579→  level: string | null;
   580→  thumbnail_url: string | null;
   581→  tags: string | null;
   582→}

   585→  id: string;
   586→  name: string;
   587→  bio: string | null;
   588→  specialization: string | null;
   589→  avatar_url: string | null;
   590→}

   592→interface SearchVideoResult {
   593→  id: string;
   594→  title: string;
   595→  description: string | null;
   596→  course_id: string | null;
   597→  duration: number | null;
   598→  sort_order: number | null;
   599→  course_title: string | null;

   602→/**
   603→ * Core search logic shared by GET /search and POST /ai-search.
   604→ * Searches courses, instructors, and videos with case-insensitive LIKE matching.
   605→ * If Workers AI binding is available, uses it to re-rank results by relevance.
   606→ */
   607→async function performSearch(c: any, query: string) {
   608→  const searchTerm = `%${query}%`;
   609→  const limit = 10;
   610→
   611→  // Run all three searches in parallel
   612→  const [coursesResult, instructorsResult, videosResult] = await Promise.all([
   613→    c.env.DB.prepare(
   614→      `SELECT id, title, description, technology_id, instructor_id, level, thumbnail_url, tags
   615→       FROM courses
   616→       WHERE is_published = 1
   617→         AND (title LIKE ? OR description LIKE ? OR tags LIKE ?)
   618→       LIMIT ?`
   619→    ).bind(searchTerm, searchTerm, searchTerm, limit).all() as { results: SearchCourseResult[] },
   620→
   621→    c.env.DB.prepare(
   622→      `SELECT id, name, bio, specialization, avatar_url
   623→       FROM instructors
   624→       WHERE is_active = 1
   625→         AND (name LIKE ? OR specialization LIKE ? OR bio LIKE ?)
   626→       LIMIT ?`
   627→    ).bind(searchTerm, searchTerm, searchTerm, limit).all() as { results: SearchInstructorResult[] },
   628→
   629→    c.env.DB.prepare(
   630→      `SELECT v.id, v.title, v.description, v.course_id, v.duration, v.sort_order,
   631→              c.title AS course_title
   632→       FROM videos v
   633→       LEFT JOIN courses c ON v.course_id = c.id
   634→       WHERE c.is_published = 1
   635→         AND (v.title LIKE ? OR v.description LIKE ?)
   636→       LIMIT ?`
   637→    ).bind(searchTerm, searchTerm, limit).all() as { results: SearchVideoResult[] },
   638→  ]);
   639→
   640→  let courses = coursesResult.results;
   641→  let instructors = instructorsResult.results;
   642→  let videos = videosResult.results;
   643→
   644→  // If Workers AI binding is available, use it to re-rank results by relevance
   645→  if (c.env.AI && (courses.length + instructors.length + videos.length) > 0) {
   646→    try {
   647→      const aiRanked = await aiReRank(c, query, courses, instructors, videos);
   648→      courses = aiRanked.courses;
   649→      instructors = aiRanked.instructors;
   650→      videos = aiRanked.videos;
   651→    } catch {
   652→      // AI ranking failed — fall back to SQL order (already good enough)
   653→    }
   654→  }
   655→
   656→  return {
   657→    query,
   658→    courses,
   659→    instructors,
   660→    videos,
   661→    total: courses.length + instructors.length + videos.length,
   662→    aiEnhanced: !!c.env.AI,
   663→  };
   664→}
   665→
   666→/**
   667→ * Use Workers AI to re-rank search results by semantic relevance.
   668→ * Sends a prompt asking the model to score each result and returns sorted arrays.
   669→ */
   670→async function aiReRank(
   671→  c: any,
   672→  query: string,
   673→  courses: SearchCourseResult[],
   674→  instructors: SearchInstructorResult[],
   675→  videos: SearchVideoResult[]
   676→): Promise<{ courses: SearchCourseResult[]; instructors: SearchInstructorResult[]; videos: SearchVideoResult[] }> {
   677→  const buildList = (items: { id: string; title?: string; name?: string; description?: string | null }[]) =>
   678→    items.map((item, i) => `${i + 1}. [id:${item.id}] ${item.title || item.name} — ${(item.description || '').slice(0, 100)}`).join('\n');
   679→
   680→  const prompt = `You are a search relevance ranker. Given the search query "${query}", assign a relevance score (0-100) to each result.
   681→
   682→COURSES:
   683→${buildList(courses.map(c => ({ id: c.id, title: c.title, description: c.description })))}
   684→
   685→INSTRUCTORS:
   686→${buildList(instructors.map(i => ({ id: i.id, name: i.name, description: i.specialization || i.bio })))}
   687→
   688→VIDEOS:
   689→${buildList(videos.map(v => ({ id: v.id, title: v.title, description: v.description })))}
   690→
   691→Respond ONLY with a JSON object like:
   692→{"courses":[{"id":"...","score":90}],"instructors":[{"id":"...","score":80}],"videos":[{"id":"...","score":70}]}`;
   693→
   694→  const aiResponse = await c.env.AI.run('@cf/meta/llama-3-8b-instruct', {
   695→    messages: [{ role: 'user', content: prompt }],
   696→    max_tokens: 512,
   697→  });
   698→
   699→  const text = (aiResponse as any)?.response || '';
   700→  // Extract JSON from the response (may be wrapped in markdown code blocks)
   701→  const jsonMatch = text.match(/\{[\s\S]*\}/);
   702→  if (!jsonMatch) throw new Error('AI did not return valid JSON');
   703→
   704→  const scores = JSON.parse(jsonMatch[0]);
   705→
   706→  const sortByScore = <T extends { id: string }>(items: T[], scoreList: { id: string; score: number }[]): T[] => {
   707→    const scoreMap = new Map(scoreList.map(s => [s.id, s.score]));
   708→    return [...items].sort((a, b) => (scoreMap.get(b.id) || 0) - (scoreMap.get(a.id) || 0));
   709→  };
   710→
   711→  return {
   712→    courses: scores.courses ? sortByScore(courses, scores.courses) : courses,
   713→    instructors: scores.instructors ? sortByScore(instructors, scores.instructors) : instructors,
   714→    videos: scores.videos ? sortByScore(videos, scores.videos) : videos,
   715→  };
   716→}
