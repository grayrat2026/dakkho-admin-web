    69→@OptIn(ExperimentalMaterial3Api::class)
    70→@Composable
    71→fun InstructorProfileScreen(
    72→    onBackClick: () -> Unit,
    73→    onNavigateToCourse: (String) -> Unit,
    74→    viewModel: InstructorProfileViewModel = hiltViewModel()
    75→) {
    76→    val uiState by viewModel.uiState.collectAsStateWithLifecycle()
    77→    val scrollBehavior = TopAppBarDefaults.pinnedScrollBehavior()
    78→
    79→    AnimatedPage {
    80→        Scaffold(
    81→            modifier = Modifier.nestedScroll(scrollBehavior.nestedScrollConnection),
    82→            topBar = {
    83→                TopAppBar(
    84→                    title = {
    85→                        Text(
    86→                            text = uiState.instructor?.name ?: "Instructor Profile",
    87→                            style = MaterialTheme.typography.titleLarge.copy(
    88→                                fontWeight = FontWeight.Bold
    89→                            ),
    90→                            maxLines = 1,
    91→                            overflow = TextOverflow.Ellipsis
    92→                        )
    93→                    },
    94→                    navigationIcon = {
    95→                        IconButton(onClick = onBackClick) {
    96→                            Icon(
    97→                                painter = painterResource(id = android.R.drawable.ic_menu_revert),
    98→                                contentDescription = "Back"
    99→                            )
   100→                        }
   101→                    },
   102→                    scrollBehavior = scrollBehavior
   103→                )
   104→            }
   105→        ) { paddingValues ->
   106→            when {
   107→                uiState.isLoading -> {
   108→                    ShimmerInstructorProfile(modifier = Modifier.padding(paddingValues))
   109→                }
   110→                uiState.error != null && uiState.instructor == null -> {
   111→                    EmptyState(
   112→                        icon = painterResource(id = android.R.drawable.ic_dialog_alert),
   113→                        title = "Could not load profile",
   114→                        subtitle = uiState.error,
   115→                        onActionClick = { viewModel.loadInstructor() },
   116→                        actionLabel = "Retry",
   117→                        modifier = Modifier.padding(paddingValues)
   118→                    )
   119→                }
   120→                uiState.instructor != null -> {
   121→                    InstructorProfileContent(
   122→                        instructor = uiState.instructor!!,
   123→                        courses = uiState.courses,
   124→                        selectedTab = uiState.selectedTab,
   125→                        onSelectTab = viewModel::selectTab,
   126→                        onCourseClick = onNavigateToCourse,
   127→                        isLoadingCourses = uiState.isLoadingCourses,
   128→                        hasMoreCourses = uiState.hasMoreCourses,
   129→                        onLoadMoreCourses = viewModel::loadMoreCourses,
   130→                        modifier = Modifier.padding(paddingValues)
   131→                    )
   132→                }
   133→            }
   134→        }
   135→    }
   136→}

   138→@Composable
   139→fun InstructorProfileContent(
   140→    instructor: InstructorDetail,
   141→    courses: List<Course>,
   142→    selectedTab: Int,
   143→    onSelectTab: (Int) -> Unit,
   144→    onCourseClick: (String) -> Unit,
   145→    isLoadingCourses: Boolean,
   146→    hasMoreCourses: Boolean,
   147→    onLoadMoreCourses: () -> Unit,
   148→    modifier: Modifier = Modifier
   149→) {
   150→    val tabs = listOf("About", "Courses")
   151→    val listState = rememberLazyListState()
   152→
   153→    val shouldLoadMore by remember {
   154→        derivedStateOf {
   155→            val lastVisibleItem = listState.layoutInfo.visibleItemsInfo.lastOrNull()?.index ?: 0
   156→            val totalItems = listState.layoutInfo.totalItemsCount
   157→            lastVisibleItem >= totalItems - 3
   158→        }
   159→    }
   160→
   161→    LaunchedEffect(shouldLoadMore) {
   162→        if (shouldLoadMore && selectedTab == 1) {
   163→            onLoadMoreCourses()
   164→        }
   165→    }
   166→
   167→    LazyColumn(
   168→        state = listState,
   169→        modifier = modifier.fillMaxSize(),
   170→        contentPadding = PaddingValues(horizontal = DesignToken.Space.dp16, vertical = DesignToken.Space.dp8)
   171→    ) {
   172→        // ── Cover + Avatar Header ──
   173→        item {
   174→            InstructorProfileHeader(instructor = instructor)
   175→        }
   176→
   177→        // ── Stats Row ──
   178→        item {
   179→            InstructorStatsRow(instructor = instructor)
   180→        }
   181→
   182→        // ── Bio Section (always visible, truncated in About tab) ──
   183→        if (!instructor.bio.isNullOrBlank() && selectedTab == 0) {
   184→            item {
   185→                Spacer(modifier = Modifier.height(DesignToken.Space.dp8))
   186→                InstructorBioSection(instructor = instructor)
   187→            }
   188→        }
   189→
   190→        // ── Social Links ──
   191→        if (instructor.socialLinks.hasAny && selectedTab == 0) {
   192→            item {
   193→                Spacer(modifier = Modifier.height(DesignToken.Space.dp12))
   194→                InstructorSocialLinks(socialLinks = instructor.socialLinks)
   195→            }
   196→        }
   197→
   198→        // ── Tab Row ──
   199→        item {
   200→            Spacer(modifier = Modifier.height(DesignToken.Space.dp16))
   201→            TabRow(
   202→                selectedTabIndex = selectedTab,
   203→                containerColor = Color.Transparent,
   204→                contentColor = MaterialTheme.colorScheme.onSurface
   205→            ) {
   206→                tabs.forEachIndexed { index, title ->
   207→                    Tab(
   208→                        selected = selectedTab == index,
   209→                        onClick = { onSelectTab(index) },
   210→                        text = {
   211→                            Text(
   212→                                text = if (index == 1) "$title (${instructor.courseCount})" else title,
   213→                                fontWeight = if (selectedTab == index) FontWeight.Bold else FontWeight.Normal
   214→                            )
   215→                        }
   216→                    )
   217→                }
   218→            }
   219→        }
   220→
   221→        // ── Tab Content ──
   222→        if (selectedTab == 0) {
   223→            // About Tab
   224→            if (!instructor.bio.isNullOrBlank()) {
   225→                item {
   226→                    Spacer(modifier = Modifier.height(DesignToken.Space.dp12))
   227→                    Text(
   228→                        text = instructor.bio,
   229→                        style = MaterialTheme.typography.bodyMedium,
   230→                        color = MaterialTheme.colorScheme.onSurface.copy(alpha = 0.85f),
   231→                        lineHeight = 22.sp
   232→                    )
   233→                }
   234→            }
   235→
   236→            if (!instructor.specialization.isNullOrBlank()) {
   237→                item {
   238→                    Spacer(modifier = Modifier.height(DesignToken.Space.dp12))
   239→                    Column {
   240→                        Text(
   241→                            text = "Specialization",
   242→                            style = MaterialTheme.typography.titleSmall.copy(
   243→                                fontWeight = FontWeight.Bold
   244→                            ),
   245→                            color = MaterialTheme.colorScheme.onSurface
   246→                        )
   247→                        Spacer(modifier = Modifier.height(DesignToken.Space.dp4))
   248→                        Text(
   249→                            text = instructor.specialization,
   250→                            style = MaterialTheme.typography.bodyMedium,
   251→                            color = SkyBlue
   252→                        )
   253→                    }
   254→                }
   255→            }
   256→
   257→            // Joined date
   258→            if (!instructor.createdAt.isNullOrBlank()) {
   259→                item {
   260→                    Spacer(modifier = Modifier.height(DesignToken.Space.dp12))
   261→                    Row(
   262→                        verticalAlignment = Alignment.CenterVertically
   263→                    ) {
   264→                        Icon(
   265→                            painter = painterResource(id = android.R.drawable.ic_menu_today),
   266→                            contentDescription = "Joined",
   267→                            tint = Neutral500,
   268→                            modifier = Modifier.size(16.dp)
   269→                        )
   270→                        Spacer(modifier = Modifier.width(DesignToken.Space.dp4))
   271→                        Text(
   272→                            text = "Joined ${formatDate(instructor.createdAt)}",
   273→                            style = MaterialTheme.typography.bodySmall,
   274→                            color = Neutral500
   275→                        )
   276→                    }
   277→                }
   278→            }
   279→        } else {
   280→            // Courses Tab
   281→            if (courses.isEmpty() && !isLoadingCourses) {
   282→                item {
   283→                    Spacer(modifier = Modifier.height(DesignToken.Space.dp32))
   284→                    EmptyState(
   285→                        icon = painterResource(id = android.R.drawable.ic_menu_gallery),
   286→                        title = "No courses yet",
   287→                        subtitle = "This instructor hasn't published any courses"
   288→                    )
   289→                }
   290→            } else {
   291→                items(courses, key = { it.id }) { course ->
   292→                    Spacer(modifier = Modifier.height(DesignToken.Space.dp8))
   293→                    InstructorCourseCard(
   294→                        course = course,
   295→                        onClick = { onCourseClick(course.id) }
   296→                    )
   297→                }
   298→
   299→                if (isLoadingCourses) {
   300→                    item {
   301→                        Box(
   302→                            modifier = Modifier
   303→                                .fillMaxWidth()
   304→                                .padding(DesignToken.Space.dp16),
   305→                            contentAlignment = Alignment.Center
   306→                        ) {
   307→                            CircularProgressIndicator(
   308→                                modifier = Modifier.size(24.dp),
   309→                                strokeWidth = 2.dp,
   310→                                color = SkyBlue
   311→                            )
   312→                        }
   313→                    }
   314→                }
   315→            }
   316→        }
   317→
   318→        // Bottom spacer
   319→        item {
   320→            Spacer(modifier = Modifier.height(DesignToken.Space.dp24))
   321→        }
   322→    }
   323→}

   325→@Composable
   326→fun InstructorProfileHeader(
   327→    instructor: InstructorDetail,
   328→    modifier: Modifier = Modifier
   329→) {
   330→    Column(
   331→        modifier = modifier.fillMaxWidth(),
   332→        horizontalAlignment = Alignment.CenterHorizontally
   333→    ) {
   334→        // Cover image or gradient
   335→        Box(
   336→            modifier = Modifier
   337→                .fillMaxWidth()
   338→                .height(120.dp)
   339→                .clip(RoundedCornerShape(DesignToken.Space.dp12)),
   340→            contentAlignment = Alignment.Center
   341→        ) {
   342→            if (instructor.coverUrl != null) {
   343→                AsyncImage(
   344→                    model = instructor.coverUrl,
   345→                    contentDescription = "Cover",
   346→                    contentScale = ContentScale.Crop,
   347→                    modifier = Modifier.fillMaxSize()
   348→                )
   349→            } else {
   350→                // Gradient fallback
   351→                Box(
   352→                    modifier = Modifier
   353→                        .fillMaxSize()
   354→                        .background(
   355→                            brush = Brush.horizontalGradient(
   356→                                colors = listOf(
   357→                                    SkyBlue.copy(alpha = 0.3f),
   358→                                    SkyBlue.copy(alpha = 0.1f),
   359→                                    MaterialTheme.colorScheme.primary.copy(alpha = 0.2f)
   360→                                )
   361→                            )
   362→                        )
   363→                )
   364→            }
   365→
   366→            // Avatar overlapping cover
   367→            // Avatar is rendered below the cover in the Column
   368→        }
   369→
   370→        Spacer(modifier = Modifier.height(DesignToken.Space.dp8))
   371→
   372→        // Avatar
   373→        Box(
   374→            modifier = Modifier.size(80.dp),
   375→            contentAlignment = Alignment.Center
   376→        ) {
   377→            if (instructor.avatarUrl != null) {
   378→                AsyncImage(
   379→                    model = instructor.avatarUrl,
   380→                    contentDescription = instructor.name,
   381→                    contentScale = ContentScale.Crop,
   382→                    modifier = Modifier
   383→                        .size(80.dp)
   384→                        .clip(CircleShape)
   385→                )
   386→            } else {
   387→                Box(
   388→                    modifier = Modifier
   389→                        .size(80.dp)
   390→                        .clip(CircleShape)
   391→                        .background(SkyBlue.copy(alpha = 0.15f)),
   392→                    contentAlignment = Alignment.Center
   393→                ) {
   394→                    Text(
   395→                        text = instructor.name.take(1).uppercase(),
   396→                        style = MaterialTheme.typography.displaySmall.copy(
   397→                            fontWeight = FontWeight.Bold
   398→                        ),
   399→                        color = SkyBlue
   400→                    )
   401→                }
   402→            }
   403→        }
   404→
   405→        Spacer(modifier = Modifier.height(DesignToken.Space.dp8))
   406→
   407→        // Name
   408→        Text(
   409→            text = instructor.name,
   410→            style = MaterialTheme.typography.headlineSmall.copy(
   411→                fontWeight = FontWeight.Bold
   412→            ),
   413→            color = MaterialTheme.colorScheme.onSurface
   414→        )
   415→
   416→        // Title / Specialization
   417→        if (instructor.title != null) {
   418→            Spacer(modifier = Modifier.height(2.dp))
   419→            Text(
   420→                text = instructor.title,
   421→                style = MaterialTheme.typography.bodyMedium,
   422→                color = SkyBlue
   423→            )
   424→        }
   425→
   426→        // Rating
   427→        Row(
   428→            verticalAlignment = Alignment.CenterVertically,
   429→            modifier = Modifier.padding(top = DesignToken.Space.dp4)
   430→        ) {
   431→            Icon(
   432→                painter = painterResource(id = android.R.drawable.btn_star_big_on),
   433→                contentDescription = "Rating",
   434→                tint = Warning,
   435→                modifier = Modifier.size(16.dp)
   436→            )
   437→            Spacer(modifier = Modifier.width(4.dp))
   438→            Text(
   439→                text = String.format("%.1f", instructor.rating),
   440→                style = MaterialTheme.typography.bodyMedium.copy(
   441→                    fontWeight = FontWeight.Bold
   442→                ),
   443→                color = MaterialTheme.colorScheme.onSurface
   444→            )
   445→        }
   446→    }
   447→}

   782→private fun formatStudentCount(count: Int): String {
   783→    return when {
   784→        count >= 1000 -> String.format("%.1fk", count / 1000f)
   785→        else -> count.toString()
   786→    }

   789→private fun formatDate(dateStr: String): String {
   790→    return try {
   791→        // Handle ISO date format: 2026-01-15T10:30:00 or 2026-01-15
   792→        val datePart = dateStr.substringBefore("T").substringBefore(" ")
   793→        val parts = datePart.split("-")
   794→        if (parts.size == 3) {
   795→            val months = listOf("", "Jan", "Feb", "Mar", "Apr", "May", "Jun",
   796→                "Jul", "Aug", "Sep", "Oct", "Nov", "Dec")
   797→            val monthIndex = parts[1].toIntOrNull() ?: 1
   798→            "${months.getOrElse(monthIndex) { "Jan" }} ${parts[0]}"
   799→        } else {
   800→            dateStr
   801→        }
   802→    } catch (e: Exception) {
   803→        dateStr
   804→    }
   805→}
