    68→@OptIn(ExperimentalMaterial3Api::class)
    69→@Composable
    70→fun InstructorProfileScreen(
    71→    onBackClick: () -> Unit,
    72→    onNavigateToCourse: (String) -> Unit,
    73→    viewModel: InstructorProfileViewModel = hiltViewModel()
    74→) {
    75→    val uiState by viewModel.uiState.collectAsStateWithLifecycle()
    76→    val scrollBehavior = TopAppBarDefaults.pinnedScrollBehavior()
    77→
    78→    AnimatedPage {
    79→        Scaffold(
    80→            modifier = Modifier.nestedScroll(scrollBehavior.nestedScrollConnection),
    81→            topBar = {
    82→                TopAppBar(
    83→                    title = {
    84→                        Text(
    85→                            text = uiState.instructor?.name ?: "Instructor Profile",
    86→                            style = MaterialTheme.typography.titleLarge.copy(
    87→                                fontWeight = FontWeight.Bold
    88→                            ),
    89→                            maxLines = 1,
    90→                            overflow = TextOverflow.Ellipsis
    91→                        )
    92→                    },
    93→                    navigationIcon = {
    94→                        IconButton(onClick = onBackClick) {
    95→                            Icon(
    96→                                painter = painterResource(id = android.R.drawable.ic_menu_revert),
    97→                                contentDescription = "Back"
    98→                            )
    99→                        }
   100→                    },
   101→                    scrollBehavior = scrollBehavior
   102→                )
   103→            }
   104→        ) { paddingValues ->
   105→            when {
   106→                uiState.isLoading -> {
   107→                    ShimmerInstructorProfile(modifier = Modifier.padding(paddingValues))
   108→                }
   109→                uiState.error != null && uiState.instructor == null -> {
   110→                    EmptyState(
   111→                        icon = painterResource(id = android.R.drawable.ic_dialog_alert),
   112→                        title = "Could not load profile",
   113→                        subtitle = uiState.error,
   114→                        onActionClick = { viewModel.loadInstructor() },
   115→                        actionLabel = "Retry",
   116→                        modifier = Modifier.padding(paddingValues)
   117→                    )
   118→                }
   119→                uiState.instructor != null -> {
   120→                    InstructorProfileContent(
   121→                        instructor = uiState.instructor!!,
   122→                        courses = uiState.courses,
   123→                        selectedTab = uiState.selectedTab,
   124→                        onSelectTab = viewModel::selectTab,
   125→                        onCourseClick = onNavigateToCourse,
   126→                        isLoadingCourses = uiState.isLoadingCourses,
   127→                        hasMoreCourses = uiState.hasMoreCourses,
   128→                        onLoadMoreCourses = viewModel::loadMoreCourses,
   129→                        modifier = Modifier.padding(paddingValues)
   130→                    )
   131→                }
   132→            }
   133→        }
   134→    }
   135→}

   137→@Composable
   138→fun InstructorProfileContent(
   139→    instructor: InstructorDetail,
   140→    courses: List<Course>,
   141→    selectedTab: Int,
   142→    onSelectTab: (Int) -> Unit,
   143→    onCourseClick: (String) -> Unit,
   144→    isLoadingCourses: Boolean,
   145→    hasMoreCourses: Boolean,
   146→    onLoadMoreCourses: () -> Unit,
   147→    modifier: Modifier = Modifier
   148→) {
   149→    val tabs = listOf("About", "Courses")
   150→    val listState = rememberLazyListState()
   151→
   152→    val shouldLoadMore by remember {
   153→        derivedStateOf {
   154→            val lastVisibleItem = listState.layoutInfo.visibleItemsInfo.lastOrNull()?.index ?: 0
   155→            val totalItems = listState.layoutInfo.totalItemsCount
   156→            lastVisibleItem >= totalItems - 3
   157→        }
   158→    }
   159→
   160→    LaunchedEffect(shouldLoadMore) {
   161→        if (shouldLoadMore && selectedTab == 1) {
   162→            onLoadMoreCourses()
   163→        }
   164→    }
   165→
   166→    LazyColumn(
   167→        state = listState,
   168→        modifier = modifier.fillMaxSize(),
   169→        contentPadding = PaddingValues(horizontal = DesignToken.Space.dp16, vertical = DesignToken.Space.dp8)
   170→    ) {
   171→        // ── Cover + Avatar Header ──
   172→        item {
   173→            InstructorProfileHeader(instructor = instructor)
   174→        }
   175→
   176→        // ── Stats Row ──
   177→        item {
   178→            InstructorStatsRow(instructor = instructor)
   179→        }
   180→
   181→        // ── Bio Section (always visible, truncated in About tab) ──
   182→        if (!instructor.bio.isNullOrBlank() && selectedTab == 0) {
   183→            item {
   184→                Spacer(modifier = Modifier.height(DesignToken.Space.dp8))
   185→                InstructorBioSection(instructor = instructor)
   186→            }
   187→        }
   188→
   189→        // ── Social Links ──
   190→        if (instructor.socialLinks.hasAny && selectedTab == 0) {
   191→            item {
   192→                Spacer(modifier = Modifier.height(DesignToken.Space.dp12))
   193→                InstructorSocialLinks(socialLinks = instructor.socialLinks)
   194→            }
   195→        }
   196→
   197→        // ── Tab Row ──
   198→        item {
   199→            Spacer(modifier = Modifier.height(DesignToken.Space.dp16))
   200→            TabRow(
   201→                selectedTabIndex = selectedTab,
   202→                containerColor = Color.Transparent,
   203→                contentColor = MaterialTheme.colorScheme.onSurface
   204→            ) {
   205→                tabs.forEachIndexed { index, title ->
   206→                    Tab(
   207→                        selected = selectedTab == index,
   208→                        onClick = { onSelectTab(index) },
   209→                        text = {
   210→                            Text(
   211→                                text = if (index == 1) "$title (${instructor.courseCount})" else title,
   212→                                fontWeight = if (selectedTab == index) FontWeight.Bold else FontWeight.Normal
   213→                            )
   214→                        }
   215→                    )
   216→                }
   217→            }
   218→        }
   219→
   220→        // ── Tab Content ──
   221→        if (selectedTab == 0) {
   222→            // About Tab
   223→            if (!instructor.bio.isNullOrBlank()) {
   224→                item {
   225→                    Spacer(modifier = Modifier.height(DesignToken.Space.dp12))
   226→                    Text(
   227→                        text = instructor.bio,
   228→                        style = MaterialTheme.typography.bodyMedium,
   229→                        color = MaterialTheme.colorScheme.onSurface.copy(alpha = 0.85f),
   230→                        lineHeight = 22.sp
   231→                    )
   232→                }
   233→            }
   234→
   235→            if (!instructor.specialization.isNullOrBlank()) {
   236→                item {
   237→                    Spacer(modifier = Modifier.height(DesignToken.Space.dp12))
   238→                    Column {
   239→                        Text(
   240→                            text = "Specialization",
   241→                            style = MaterialTheme.typography.titleSmall.copy(
   242→                                fontWeight = FontWeight.Bold
   243→                            ),
   244→                            color = MaterialTheme.colorScheme.onSurface
   245→                        )
   246→                        Spacer(modifier = Modifier.height(DesignToken.Space.dp4))
   247→                        Text(
   248→                            text = instructor.specialization,
   249→                            style = MaterialTheme.typography.bodyMedium,
   250→                            color = SkyBlue
   251→                        )
   252→                    }
   253→                }
   254→            }
   255→
   256→            // Joined date
   257→            if (!instructor.createdAt.isNullOrBlank()) {
   258→                item {
   259→                    Spacer(modifier = Modifier.height(DesignToken.Space.dp12))
   260→                    Row(
   261→                        verticalAlignment = Alignment.CenterVertically
   262→                    ) {
   263→                        Icon(
   264→                            painter = painterResource(id = android.R.drawable.ic_menu_today),
   265→                            contentDescription = "Joined",
   266→                            tint = Neutral500,
   267→                            modifier = Modifier.size(16.dp)
   268→                        )
   269→                        Spacer(modifier = Modifier.width(DesignToken.Space.dp4))
   270→                        Text(
   271→                            text = "Joined ${formatDate(instructor.createdAt)}",
   272→                            style = MaterialTheme.typography.bodySmall,
   273→                            color = Neutral500
   274→                        )
   275→                    }
   276→                }
   277→            }
   278→        } else {
   279→            // Courses Tab
   280→            if (courses.isEmpty() && !isLoadingCourses) {
   281→                item {
   282→                    Spacer(modifier = Modifier.height(DesignToken.Space.dp32))
   283→                    EmptyState(
   284→                        icon = painterResource(id = android.R.drawable.ic_menu_gallery),
   285→                        title = "No courses yet",
   286→                        subtitle = "This instructor hasn't published any courses"
   287→                    )
   288→                }
   289→            } else {
   290→                items(courses, key = { it.id }) { course ->
   291→                    Spacer(modifier = Modifier.height(DesignToken.Space.dp8))
   292→                    InstructorCourseCard(
   293→                        course = course,
   294→                        onClick = { onCourseClick(course.id) }
   295→                    )
   296→                }
   297→
   298→                if (isLoadingCourses) {
   299→                    item {
   300→                        Box(
   301→                            modifier = Modifier
   302→                                .fillMaxWidth()
   303→                                .padding(DesignToken.Space.dp16),
   304→                            contentAlignment = Alignment.Center
   305→                        ) {
   306→                            CircularProgressIndicator(
   307→                                modifier = Modifier.size(24.dp),
   308→                                strokeWidth = 2.dp,
   309→                                color = SkyBlue
   310→                            )
   311→                        }
   312→                    }
   313→                }
   314→            }
   315→        }
   316→
   317→        // Bottom spacer
   318→        item {
   319→            Spacer(modifier = Modifier.height(DesignToken.Space.dp24))
   320→        }
   321→    }
   322→}

   324→@Composable
   325→fun InstructorProfileHeader(
   326→    instructor: InstructorDetail,
   327→    modifier: Modifier = Modifier
   328→) {
   329→    Column(
   330→        modifier = modifier.fillMaxWidth(),
   331→        horizontalAlignment = Alignment.CenterHorizontally
   332→    ) {
   333→        // Cover image or gradient
   334→        Box(
   335→            modifier = Modifier
   336→                .fillMaxWidth()
   337→                .height(120.dp)
   338→                .clip(RoundedCornerShape(DesignToken.Space.dp12)),
   339→            contentAlignment = Alignment.Center
   340→        ) {
   341→            if (instructor.coverUrl != null) {
   342→                AsyncImage(
   343→                    model = instructor.coverUrl,
   344→                    contentDescription = "Cover",
   345→                    contentScale = ContentScale.Crop,
   346→                    modifier = Modifier.fillMaxSize()
   347→                )
   348→            } else {
   349→                // Gradient fallback
   350→                Box(
   351→                    modifier = Modifier
   352→                        .fillMaxSize()
   353→                        .background(
   354→                            brush = Brush.horizontalGradient(
   355→                                colors = listOf(
   356→                                    SkyBlue.copy(alpha = 0.3f),
   357→                                    SkyBlue.copy(alpha = 0.1f),
   358→                                    MaterialTheme.colorScheme.primary.copy(alpha = 0.2f)
   359→                                )
   360→                            )
   361→                        )
   362→                )
   363→            }
   364→
   365→            // Avatar overlapping cover
   366→            // Avatar is rendered below the cover in the Column
   367→        }
   368→
   369→        Spacer(modifier = Modifier.height(DesignToken.Space.dp8))
   370→
   371→        // Avatar
   372→        Box(
   373→            modifier = Modifier.size(80.dp),
   374→            contentAlignment = Alignment.Center
   375→        ) {
   376→            if (instructor.avatarUrl != null) {
   377→                AsyncImage(
   378→                    model = instructor.avatarUrl,
   379→                    contentDescription = instructor.name,
   380→                    contentScale = ContentScale.Crop,
   381→                    modifier = Modifier
   382→                        .size(80.dp)
   383→                        .clip(CircleShape)
   384→                )
   385→            } else {
   386→                Box(
   387→                    modifier = Modifier
   388→                        .size(80.dp)
   389→                        .clip(CircleShape)
   390→                        .background(SkyBlue.copy(alpha = 0.15f)),
   391→                    contentAlignment = Alignment.Center
   392→                ) {
   393→                    Text(
   394→                        text = instructor.name.take(1).uppercase(),
   395→                        style = MaterialTheme.typography.displaySmall.copy(
   396→                            fontWeight = FontWeight.Bold
   397→                        ),
   398→                        color = SkyBlue
   399→                    )
   400→                }
   401→            }
   402→        }
   403→
   404→        Spacer(modifier = Modifier.height(DesignToken.Space.dp8))
   405→
   406→        // Name
   407→        Text(
   408→            text = instructor.name,
   409→            style = MaterialTheme.typography.headlineSmall.copy(
   410→                fontWeight = FontWeight.Bold
   411→            ),
   412→            color = MaterialTheme.colorScheme.onSurface
   413→        )
   414→
   415→        // Title / Specialization
   416→        if (instructor.title != null) {
   417→            Spacer(modifier = Modifier.height(2.dp))
   418→            Text(
   419→                text = instructor.title,
   420→                style = MaterialTheme.typography.bodyMedium,
   421→                color = SkyBlue
   422→            )
   423→        }
   424→
   425→        // Rating
   426→        Row(
   427→            verticalAlignment = Alignment.CenterVertically,
   428→            modifier = Modifier.padding(top = DesignToken.Space.dp4)
   429→        ) {
   430→            Icon(
   431→                painter = painterResource(id = android.R.drawable.btn_star_big_on),
   432→                contentDescription = "Rating",
   433→                tint = Warning,
   434→                modifier = Modifier.size(16.dp)
   435→            )
   436→            Spacer(modifier = Modifier.width(4.dp))
   437→            Text(
   438→                text = String.format("%.1f", instructor.rating),
   439→                style = MaterialTheme.typography.bodyMedium.copy(
   440→                    fontWeight = FontWeight.Bold
   441→                ),
   442→                color = MaterialTheme.colorScheme.onSurface
   443→            )
   444→        }
   445→    }
   446→}

   781→private fun formatStudentCount(count: Int): String {
   782→    return when {
   783→        count >= 1000 -> String.format("%.1fk", count / 1000f)
   784→        else -> count.toString()
   785→    }

   788→private fun formatDate(dateStr: String): String {
   789→    return try {
   790→        // Handle ISO date format: 2026-01-15T10:30:00 or 2026-01-15
   791→        val datePart = dateStr.substringBefore("T").substringBefore(" ")
   792→        val parts = datePart.split("-")
   793→        if (parts.size == 3) {
   794→            val months = listOf("", "Jan", "Feb", "Mar", "Apr", "May", "Jun",
   795→                "Jul", "Aug", "Sep", "Oct", "Nov", "Dec")
   796→            val monthIndex = parts[1].toIntOrNull() ?: 1
   797→            "${months.getOrElse(monthIndex) { "Jan" }} ${parts[0]}"
   798→        } else {
   799→            dateStr
   800→        }
   801→    } catch (e: Exception) {
   802→        dateStr
   803→    }
   804→}
