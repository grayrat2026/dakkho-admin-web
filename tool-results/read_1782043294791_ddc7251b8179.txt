   147→   147→    modifier: Modifier = Modifier
   148→   148→) {

   153→   153→        derivedStateOf {
   154→   154→            val lastVisibleItem = listState.layoutInfo.visibleItemsInfo.lastOrNull()?.index ?: 0

   156→   156→            lastVisibleItem >= totalItems - 3
   157→   157→        }

   159→   159→
   160→   160→    LaunchedEffect(shouldLoadMore) {
   161→   161→        if (shouldLoadMore && selectedTab == 1) {
   162→   162→            onLoadMoreCourses()

   164→   164→    }
   165→   165→
   166→   166→    LazyColumn(
   167→   167→        state = listState,
   168→   168→        modifier = modifier.fillMaxSize(),
   169→   169→        contentPadding = PaddingValues(horizontal = DesignToken.Space.dp16, vertical = DesignToken.Space.dp8)
   170→   170→    ) {
   171→   171→        // ── Cover + Avatar Header ──

   173→   173→            InstructorProfileHeader(instructor = instructor)
   174→   174→        }
   175→   175→
   176→   176→        // ── Stats Row ──
   177→   177→        item {
   178→   178→            InstructorStatsRow(instructor = instructor)
   179→   179→        }
   180→   180→
   181→   181→        // ── Bio Section (always visible, truncated in About tab) ──

   183→   183→            item {
   184→   184→                Spacer(modifier = Modifier.height(DesignToken.Space.dp8))
   185→   185→                InstructorBioSection(instructor = instructor)
   186→   186→            }
   187→   187→        }
   188→   188→
   189→   189→        // ── Social Links ──
   190→   190→        if (instructor.socialLinks.hasAny && selectedTab == 0) {
   191→   191→            item {
   192→   192→                Spacer(modifier = Modifier.height(DesignToken.Space.dp12))
   193→   193→                InstructorSocialLinks(socialLinks = instructor.socialLinks)
   194→   194→            }
   195→   195→        }
   196→   196→
   197→   197→        // ── Tab Row ──
   198→   198→        item {
   199→   199→            Spacer(modifier = Modifier.height(DesignToken.Space.dp16))
   200→   200→            TabRow(
   201→   201→                selectedTabIndex = selectedTab,
   202→   202→                containerColor = Color.Transparent,
   203→   203→                contentColor = MaterialTheme.colorScheme.onSurface
   204→   204→            ) {
   205→   205→                tabs.forEachIndexed { index, title ->
   206→   206→                    Tab(
   207→   207→                        selected = selectedTab == index,
   208→   208→                        onClick = { onSelectTab(index) },
   209→   209→                        text = {
   210→   210→                            Text(
   211→   211→                                text = if (index == 1) "$title (${instructor.courseCount})" else title,
   212→   212→                                fontWeight = if (selectedTab == index) FontWeight.Bold else FontWeight.Normal
   213→   213→                            )
   214→   214→                        }
   215→   215→                    )
   216→   216→                }
   217→   217→            }
   218→   218→        }
   219→   219→
   220→   220→        // ── Tab Content ──
   221→   221→        if (selectedTab == 0) {
   222→   222→            // About Tab
   223→   223→            if (!instructor.bio.isNullOrBlank()) {
   224→   224→                item {
   225→   225→                    Spacer(modifier = Modifier.height(DesignToken.Space.dp12))
   226→   226→                    Text(
   227→   227→                        text = instructor.bio,
   228→   228→                        style = MaterialTheme.typography.bodyMedium,
   229→   229→                        color = MaterialTheme.colorScheme.onSurface.copy(alpha = 0.85f),
   230→   230→                        lineHeight = 22.sp
   231→   231→                    )
   232→   232→                }
   233→   233→            }
   234→   234→
   235→   235→            if (!instructor.specialization.isNullOrBlank()) {
   236→   236→                item {
   237→   237→                    Spacer(modifier = Modifier.height(DesignToken.Space.dp12))
   238→   238→                    Column {
   239→   239→                        Text(
   240→   240→                            text = "Specialization",
   241→   241→                            style = MaterialTheme.typography.titleSmall.copy(
   242→   242→                                fontWeight = FontWeight.Bold
   243→   243→                            ),
   244→   244→                            color = MaterialTheme.colorScheme.onSurface
   245→   245→                        )
   246→   246→                        Spacer(modifier = Modifier.height(DesignToken.Space.dp4))
   247→   247→                        Text(
   248→   248→                            text = instructor.specialization,
   249→   249→                            style = MaterialTheme.typography.bodyMedium,
   250→   250→                            color = SkyBlue
   251→   251→                        )
   252→   252→                    }
   253→   253→                }
   254→   254→            }
   255→   255→
   256→   256→            // Joined date
   257→   257→            if (!instructor.createdAt.isNullOrBlank()) {

   260→   260→                    Row(

   264→   264→                            painter = painterResource(id = android.R.drawable.ic_menu_today),
   265→   265→                            contentDescription = "Joined",
   266→   266→                            tint = Neutral500,
   267→   267→                            modifier = Modifier.size(16.dp)
   268→   268→                        )
   269→   269→                        Spacer(modifier = Modifier.width(DesignToken.Space.dp4))
   270→   270→                        Text(
   271→   271→                            text = "Joined ${formatDate(instructor.createdAt)}",
   272→   272→                            style = MaterialTheme.typography.bodySmall,
   273→   273→                            color = Neutral500
   274→   274→                        )
   275→   275→                    }
   276→   276→                }
   277→   277→            }
   278→   278→        } else {
   279→   279→            // Courses Tab
   280→   280→            if (courses.isEmpty() && !isLoadingCourses) {
   281→   281→                item {
   282→   282→                    Spacer(modifier = Modifier.height(DesignToken.Space.dp32))
   283→   283→                    EmptyState(
   284→   284→                        icon = painterResource(id = android.R.drawable.ic_menu_gallery),
   285→   285→                        title = "No courses yet",
   286→   286→                        subtitle = "This instructor hasn't published any courses"
   287→   287→                    )
   288→   288→                }
   289→   289→            } else {
   290→   290→                items(courses, key = { it.id }) { course ->
   291→   291→                    Spacer(modifier = Modifier.height(DesignToken.Space.dp8))
   292→   292→                    InstructorCourseCard(
   293→   293→                        course = course,
   294→   294→                        onClick = { onCourseClick(course.id) }
   295→   295→                    )
   296→   296→                }
   297→   297→
   298→   298→                if (isLoadingCourses) {
   299→   299→                    item {
   300→   300→                        Box(
   301→   301→                            modifier = Modifier
   302→   302→                                .fillMaxWidth()
   303→   303→                                .padding(DesignToken.Space.dp16),
   304→   304→                            contentAlignment = Alignment.Center

   306→   306→                            CircularProgressIndicator(
   307→   307→                                modifier = Modifier.size(24.dp),

   321→   321→    }

   788→   788→private fun formatDate(dateStr: String): String {
   789→   789→    return try {
   790→   790→        // Handle ISO date format: 2026-01-15T10:30:00 or 2026-01-15
   791→   791→        val datePart = dateStr.substringBefore("T").substringBefore(" ")
   792→   792→        val parts = datePart.split("-")
   793→   793→        if (parts.size == 3) {
   794→   794→            val months = listOf("", "Jan", "Feb", "Mar", "Apr", "May", "Jun",
   795→   795→                "Jul", "Aug", "Sep", "Oct", "Nov", "Dec")
   796→   796→            val monthIndex = parts[1].toIntOrNull() ?: 1
   797→   797→            "${months.getOrElse(monthIndex) { "Jan" }} ${parts[0]}"
   798→   798→        } else {
   799→   799→            dateStr
   800→   800→        }
   801→   801→    } catch (e: Exception) {
   802→   802→        dateStr
   803→   803→    }
   804→   804→}
